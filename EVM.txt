#include<stdio.h>
#include<conio.h>
void main(){
    int a=0,b-0,c-0,iv=0,ch,nv;
    char name[20];
clrscr();
    printf("Enter pooling booth officer name:");
    gets(name);
    printf("Enter number of voters:");
    scanf("%d",&nv);
    printf("Each party initial votes:");
    printf("\nA-%d B-%d C-%d",a,b,c);
    getch();
    clrscr();
    while(nv>0)
    {
        printf("\n1.A 2.B 3.C");
        printf("Enter your vote:");
        scanf("%d",&ch);
      switch(ch)
      {
          case 1:a++;nv--;break;
          case 2:b++;nv--;break;
          case 3:c++;nv--;break;
          default:iv++;nv--;
      }
      clrscr();
      }
      printf("\nA-%d B-%d C-%d",a,b,c);
      printf("\nInvalid votes-%d",iv);
getch();
}
